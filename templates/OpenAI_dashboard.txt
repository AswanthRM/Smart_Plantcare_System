<script>
async function capture(){

document.getElementById("result").innerHTML="Capturing image...";
document.getElementById("advice").innerHTML="";
document.getElementById("photo").style.display="none";

let res = await fetch("/capture");
let data = await res.json();

if(data.error){
document.getElementById("result").innerHTML=data.error;
return;
}

document.getElementById("photo").src =
"data:image/jpeg;base64," + data.image;

document.getElementById("photo").style.display="block";

document.getElementById("result").innerHTML =
data.prediction + " (" + data.confidence + "%)";

document.getElementById("temp").innerHTML =
data.temperature + " °C";

document.getElementById("hum").innerHTML =
data.humidity + " %";
}

async function getAdvice(){

document.getElementById("advice").innerHTML =
"<div class='loading'>Generating recommendations...</div>";

let res = await fetch("/advice");
let data = await res.json();

if(data.error){
document.getElementById("advice").innerHTML=data.error;
return;
}

let txt = data.advice.replace(/\*\*/g,"");

let cause="", remedy="", prevent="";
let lines = txt.split("\n");

for(let line of lines){

if(line.toLowerCase().includes("cause")){
cause = line.replace(/1\. Cause:?/i,"").trim();
}
else if(line.toLowerCase().includes("immediate remedy")){
remedy = line.replace(/2\. Immediate remedy:?/i,"").trim();
}
else if(line.toLowerCase().includes("prevention")){
prevent = line.replace(/3\. Prevention:?/i,"").trim();
}
}

document.getElementById("advice").innerHTML = `
<div class="ai-card">

<div class="ai-head">Action Plan</div>

<div class="section">
<h3 style="color:#ff8e8e;">Cause</h3>
<p>${cause}</p>
</div>

<div class="section">
<h3 style="color:#f7c948;">Immediate Remedy</h3>
<p>${remedy}</p>
</div>

<div class="section">
<h3 style="color:#19c38a;">Prevention</h3>
<p>${prevent}</p>
</div>

</div>
`;
}
</script>
